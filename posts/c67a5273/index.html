#include <stdio.h>
fib(int g){
    switch (g)
    {
    case 0:return 0;
    case 1:
    case 2:return 1;
    }
    return(fib(g-1)+fib(g-2));
    }
void main(){
    int k;
    scanf("%d",&k);
    k=fib(k);
    printf("k=%d",k);
}
